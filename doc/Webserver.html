def Webserver::start
  app = Rack::Builder.app do

    map "/" do
      run Webserver::GC29Tester
    end

    map "/gv" do
      run GraphViewerServer
    end

  end

  
  Rack::Handler.get('webrick').run(app, :Host=>'0.0.0.0', :Port=>config(:WebServerPort) || 8080) do |server|
    # trap(:INT) { server.stop }
  end

end
