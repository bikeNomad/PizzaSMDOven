def [](key)
  if key.instance_of?(Fixnum)
    @slave.send("read_#{@type}", key, 1)
  elsif key.instance_of?(Range)
    @slave.send("read_#{@type}s", key.first, key.count)
  else
    raise ProxyException, "Invalid argument, must be integer or range. Was #{key.class}"
  end
end
