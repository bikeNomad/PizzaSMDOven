  def csv_to_object
    case self
      when /^-?(?:[0-9]+\.)?[0-9]+(?:[efgEFG][+-]?[0-9]+)?$/
        self.to_f
      when /^"(.*)"$/
        $1.gsub(/\\"/, '"')
      when "true", "TRUE"
        true
      when "nil", "NIL", "NULL"
        nil
      when "false", "FALSE"
        false
#      when /^(?:Mon|Tue|Wed|Thu|Fri|Sat|Sun) /
#        Time.parse(self) rescue self
      else
        self
    end
  end
