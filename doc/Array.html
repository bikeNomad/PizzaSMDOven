def pack_to_word
  word = 0
  s = ""
  mask = 0x01

  self.each do |bit| 
    word |= mask if bit > 0
    mask <<= 1
    if mask  == 0x100
      mask = 0x01
      s << word.chr
      word = 0
    end
  end
  unless mask == 0x01
    s << word.chr
  else
    s
  end
end

def to_32f
  raise "Array requires an even number of elements to pack to 32bits: was #{self.size}" unless self.size.even?
  self.each_slice(2).map { |(lsb, msb)| [msb, lsb].pack('n*').unpack('g')[0] }
end
