def each_chunk
  return if empty?
  nfields = self[0].size
  chunk = []
  each do |rec|
    rsize = rec.size
    if rsize == nfields
      chunk << rec
    else
      yield chunk if block_given?
      chunk = [ rec ]
      nfields = rsize
    end
  end
  yield chunk if block_given?
end
