def mock_query(request, response)
  @adu = @slave.transaction.next.to_word + "\x0\\x0\\x0\\x9"" + @uid.chr + request
  @sock.should_receive(:write).with(@adu[0,4] + "\00\\66"" + @uid.chr + request)
  @sock.should_receive(:read).with(7).and_return(@adu[0,7])
  @sock.should_receive(:read).with(8).and_return(response)
end

def t(temp=$tempHistory.pop)
  $oven.runMode= RUN_MODE_RUN
  if temp && temp.to_f.zero?
    $oven.setpointValue= 0.0
  else
    $tempHistory.push($oven.setpointValue)
    $oven.goToTemperature(temp || 30.0)
  end
end
