def update(_error, _actual, _samplePeriod)
  if @window
    if _error > @window
      return @outOfWindowOutputs[1]
    elsif _error < - @window
      return @outOfWindowOutputs[0]
    end
  end
  @lastPidOutput = pid.update(_error, _actual, _samplePeriod)
  @lastPidOutput + feedForward
end
